CREATE DATABASE INDEX1;

USE INDEX1;

SELECT * FROM CLIENTE;

-- 0. Exibir estrutura da tabela
DESC CLIENTE;
DESC TIPOCLIENTE;
DESC ARTIGO;

-- 1. EXIBIR ÍNDICE
 SHOW INDEX FROM CLIENTE;
 SHOW INDEX FROM TIPOCLIENTE;
SHOW INDEX FROM ARTIGO;
 
 -- 2. Criar índice
CREATE INDEX IDX_EMAIL ON CLIENTE(EMAIL);

-- 3. Excluir indice
DROP INDEX IDX_EMAIL ON CLIENTE;
DROP INDEX INDEX_CNPJ_NOMEFANTASIA ON CLIENTE;

-- 4. Criar índice composto
CREATE INDEX INDEX_CNPJ_NOMEFANTASIA ON CLIENTE(CNPJ,NOMEFANTASIA);


-- CNPJ: 2015000000102 - EMAIL: 0.860395061653475@gmail.com
-- EMAIL: 0.727761728435533@gmail.com

WHERE CNPJ = 2015000000102
AND EMAIL =  '0.860395061653475@gmail.com';
